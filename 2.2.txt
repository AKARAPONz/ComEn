char *reverse( char str1[ ] ) {
    int num = 0;
    while ( str1[ num ] != '\0' ) {
        num++ ;
    }

    for ( int i = 0, j = num - 1; i < j; i++, j-- ) {
        char save = str1[ i ] ;
        str1[ i ] = str1[ j ] ;
        str1[ j ] = save ;
    }

    for ( int i = 0; i < num; i++ ) {
        printf( "%c", str1[ i ] ) ;
    }

    return str1 ;
}
